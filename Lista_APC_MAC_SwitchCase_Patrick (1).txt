         break;

         break;

2. Calculadora Simples
-------------------------------------
#include <iostream>
using namespace std;

int main() {
    float num1, num2;
    char op;
    cout << "Digite dois números: ";
    cin >> num1 >> num2;
    cout << "Digite a operação (+, -, *, /): ";
    cin >> op;

    switch(op) {
        case '+':
            cout << "Resultado: " << num1 + num2 << endl;
            break;
        case '-':
            cout << "Resultado: " << num1 - num2 << endl;
            break;
        case '*':
            cout << "Resultado: " << num1 * num2 << endl;
            break;
        case '/':
            if (num2 != 0)
                cout << "Resultado: " << num1 / num2 << endl;
            else
                cout << "Divisão por zero!" << endl;
            break;
        default:
            cout << "Operação inválida!" << endl;
    }
    return 0;
}

4. Menu de Opções
---------------------------

using namespace std;

int main() {
    int opcao;
    cout << "Menu:\n1 - Novo Jogo\n2 - Carregar Jogo\n3 - Configurações\n4 - Sair\nEscolha uma opção: ";
    cin >> opcao;

    switch(opcao) {
        case 1:
            cout << "Novo Jogo iniciado!" << endl;
            break;
        case 2:
            cout << "Carregando jogo..." << endl;
            break;
        case 3:
            cout << "Abrindo configurações..." << endl;
            break;
        case 4:
            cout << "Saindo do jogo." << endl;

}


5. Conversão Mês Estação
-------------------------
#include <iostream>
using namespace std;

int main() {
    int mes;
    cout << "Digite o número do mês (1 a 12): ";
    cin >> mes;

    switch(mes) {
        case 12: case 1: case 2:
            cout << "Estação: Verão" << endl;
            break;
        case 3: case 4: case 5:
            cout << "Estação: Outono" << endl;
            break;
        case 6: case 7: case 8:
            cout << "Estação: Inverno" << endl;
            break;
        case 9: case 10: case 11:
            cout << "Estação: Primavera" << endl;
            break;
        default:
            cout << "Mês inválido!" << endl;
    }
    return 0;
}
